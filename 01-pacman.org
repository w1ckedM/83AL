** pkglist

  Operation = Install
  Operation = Remove
  Type = Package
  Target = *

  [Action]
  When = PostTransaction
  Exec = /bin/sh -c '/usr/bin/pacman -Qqe > /home/wimi/dotfile/pkglist.txt'
